import { Module } from '@nestjs/common';
import { JwtModule } from '@nestjs/jwt';
import { TypeOrmModule } from '@nestjs/typeorm';
import { Login } from 'src/entity/typeorm/login.enetiy';
import { User } from 'src/entity/typeorm/user.emtity';
import { VaccineDetails } from 'src/entity/typeorm/vaccinationData.entity';
import { VaccineType } from 'src/entity/typeorm/vaccineType.emtity';
import { LoginService } from '../login/login.service';
import { UserService } from '../user/user.service';
import { VaccineTypeService } from '../vaccine-type/vaccine-type.service';
import { VaccineDetailsController } from './vaccination-data.controller';
import { VaccineDetailsService } from './vaccination-data.service';

@Module({
  imports: [TypeOrmModule.forFeature([VaccineDetails, User, Login,VaccineType]),
  JwtModule.register({
    secret: 'secret',
    signOptions: { expiresIn: '1d' },
  })],
  controllers: [VaccineDetailsController],
  providers: [VaccineDetailsService,VaccineTypeService,UserService,LoginService]
})
export class VaccineDetailsModule { }
