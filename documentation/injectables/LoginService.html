import { BadRequestException, Injectable } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { InjectRepository } from '@nestjs/typeorm';
import { Login } from 'src/entity/typeorm/login.enetiy';
import { Repository } from 'typeorm';
import * as bcrypt from 'bcrypt';
import { Request } from 'express';
import { UserService } from '../user/user.service';


@Injectable()
export class LoginService {
    constructor(
        @InjectRepository(Login)
        private loginRepository: Repository<Login>,
        private jwtService: JwtService,
        private userService: UserService

    ) { }

    /**
     * 
     * @param login recives number as input
     * @returns a promise if login object gets added succefully
     */
    async addLoginDetails(login: Login): Promise<Login> {
        return await this.loginRepository.save(login)
    }

    /**
     * 
     * @param emailId recives string as input
     * @returns a promise if object is found
     */
    async getByEmailId(emailId: string): Promise<Login> {
        return await this.loginRepository.findOne(
            { emailId: emailId }
        )
    }

    /**
     * 
     * @param condition recives any type of input
     * @returns a promise object based on condition
     */
    async getById(condition: any): Promise<Login> {
        return this.loginRepository.findOne(condition);
    }

    /**
     * 
     * @param id recives number as input
     * @param login  recvies login obj as input
     * @returns a message if user gets deleted succesfully
     */
    async updateLoginDetails(id: number, login: Login) {
        const loginObj: Login = new Login();
        const date = new Date();
        loginObj.createTime = login.createTime;
        loginObj.createdAt = login.createdAt;
        loginObj.emailId = login.emailId;
        //loginObj.id = id;
        // loginObj.updatedAt = date.getDate().toString();
        // loginObj.updatedTime = date.getTime().toString();
        loginObj.isActive = false;
        loginObj.password = login.password;
        loginObj.updatedAt = date.toLocaleTimeString();
        loginObj.updatedTime = date.toLocaleDateString();
        // Object.assign({}, loginObj1, login, loginObj);
        console.log('login:  ', loginObj, 'id: ', id);

        return await this.loginRepository.update(id, loginObj);

    }

    /**
     * 
     * @param emailId recives string as input
     * @param password recives string as input
     * @param response return jwt token based on login object
     * @returns 
     */
    async loginJwtToken(emailId: string, password: string, response: any): Promise<any> {
        const login = await this.userService.getByEmailId(emailId);
        if (!login) {
            throw new BadRequestException('Invalid Credentials');
        }
        if (!await bcrypt.compare(password, login.password)) {
            throw new BadRequestException('invalid Credentials');
        }
        const jwt = await this.jwtService.signAsync({ id: login.id });
        console.log("';';", jwt);
        response.cookie('jwt', jwt, { httpOnly: true });
        return jwt;
    }

    result: any = "";
    /**
     * 
     * @param request recives request object as input
     * @returns return current logged in user object
     */
    async getLogggedInUser(request: Request) {
        const cookie = request.cookies['jwt'];
        console.log("cookie : ", cookie);
        const data = await this.jwtService.verifyAsync(cookie);
        console.log("data: ", data);

        // const loginObjCurrent = await this.getById({ id: data['id'] });
        // // eslint-disable-next-line @typescript-eslint/no-unused-vars
        // const { password, ...result } = loginObjCurrent;
        // this.result = result;
        return data;

    }

}
