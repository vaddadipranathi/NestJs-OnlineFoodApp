import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { HttpException, Injectable } from '@nestjs/common';
import { HttpStatus } from '@nestjs/common';
import { UserService } from '../user/user.service';
import { DataNotFoundException, UserNotFoundException } from 'src/common/filter/customException.exception';
import { AddVaccineDetailsDto } from 'src/entity/dto/addVaccineDetails.dto';
import { VaccineDetails } from 'src/entity/typeorm/vaccinationData.entity';
import { VaccineTypeService } from '../vaccine-type/vaccine-type.service';


/**
 * starting point of VaccineDetailsService
 */
@Injectable()
export class VaccineDetailsService {
    constructor(
        @InjectRepository(VaccineDetails)
        private vaccineDetailsRepository: Repository<VaccineDetails>,
        private vaccineTypeService: VaccineTypeService,
        private userService: UserService,
    ) { }
    /**
     * 
     * @param vaccineDetails accepts objects of vaccineDetails
     * @param emailId accepts emailId of type string
     * @returns the string, if vaccinedetails are added successfully
     */
    async addVaccineDetails(vaccineDetails: AddVaccineDetailsDto, emailId: string): Promise<string> {
        const result = await this.userService.getByEmailId(emailId);
        if (result) {
            const vaccDetailsObj = new VaccineDetails();
            const result2 = await this.vaccineTypeService.getVaccineType(vaccineDetails.vaccineName);
            vaccDetailsObj.user = result;
            vaccDetailsObj.isActive = true;
            const date = new Date();
            vaccDetailsObj.createTime = date.toLocaleTimeString();
            vaccDetailsObj.createdAt = date.toLocaleDateString();
            vaccDetailsObj.updatedAt = " ";
            vaccDetailsObj.updatedTime = " ";
            vaccDetailsObj.noOfDoses = vaccineDetails.noOfDoses;

            if (vaccDetailsObj.noOfDoses == 1) {
                vaccDetailsObj.status = "Parital";
                vaccDetailsObj.doseDate = vaccineDetails.doseDate;
                vaccDetailsObj.vaccineName = vaccineDetails.vaccineName;
            }
            else if (vaccDetailsObj.noOfDoses == 2) {
                vaccDetailsObj.status = "Done With Vaccination";
                vaccDetailsObj.doseDate = vaccineDetails.doseDate;
                vaccDetailsObj.vaccineName = vaccineDetails.vaccineName;
            }
            else {
                throw new HttpException("User is not vaccinated yet", HttpStatus.BAD_REQUEST);
            }
            vaccDetailsObj.vaccineType = result2;
            console.log("from ", result2);
            console.log("from resultt", result);
            await this.vaccineDetailsRepository.save(vaccDetailsObj);
            return "Added Successfully";
        }

        else {
            throw new UserNotFoundException();
        }

    }

    /**
     * 
     * @returns all the vaccinationDetails from vaccination table
     */
    async getVaccineData(): Promise<VaccineDetails[]> {
        const result = await this.vaccineDetailsRepository.find();
        if (result) {
            return result;
        }
        else {
            throw new DataNotFoundException();
        }
    }

    /**
     * 
     * @param date accepts date of type date
     * @returns the vaccineData based on the date
     */

    async getVaccineDataOnDate(date: Date): Promise<VaccineDetails[]> {
        const result = await this.vaccineDetailsRepository.find(
            { doseDate: date }
        )
        if (result) {
            return result;
        }
        else {
            throw new DataNotFoundException();
        }
    }

    async logicalUserDelete(id:number):Promise<string>{
        const userResult =await this.userService.getById(id);
        userResult.isActive=false;
        this.vaccineDetailsRepository.update(id,userResult);
        return "Deleted Successfully"
    }

}
