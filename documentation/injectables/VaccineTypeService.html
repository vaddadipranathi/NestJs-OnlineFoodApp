import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { VaccineType } from 'src/entity/typeorm/vaccineType.emtity';
import { Repository } from 'typeorm';

/**
 * starting of VaccineTypeService
 */
@Injectable()
export class VaccineTypeService {
    constructor(
        @InjectRepository(VaccineType)
        private vaccineTypeRepository: Repository<VaccineType>,

    ) { }

    /**
     * 
     * @param vaccineType accepts object of vaccineType
     * @returns a string, if details are added successfully
     */

    async addVaccineType(vaccineType:VaccineType):Promise<string>{
        await this.vaccineTypeRepository.save(vaccineType);
        return "Added Successfully";
    }
    async getVaccineType(vaccineName:string):Promise<VaccineType>{
        return  await this.vaccineTypeRepository.findOne(
            {vaccineName:vaccineName}
        )
    }
}
