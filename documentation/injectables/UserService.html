import { forwardRef, Inject, Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { AddUserDto } from 'src/entity/dto/addUser.dto';
import { Login } from 'src/entity/typeorm/login.enetiy';
import { User } from 'src/entity/typeorm/user.emtity';
import { Repository } from 'typeorm';
import { LoginService } from './../login/login.service';

/**
 * starting point of UserSevice
 */
@Injectable()
export class UserService {
    constructor(
        @InjectRepository(User)
        private userRepository: Repository<User>,
        @InjectRepository(Login)
        private loginRepository: Repository<Login>,
        //   @Inject(forwardRef(() => LoginService  private loginService: LoginService
        @Inject(forwardRef(() => LoginService)) private readonly loginService: LoginService,

    ) { }

    /**
     * 
     * @param addUserDto accepts objects of adduserDto
     * @returns user object , if user is added Successfully
     */
    async addUserDetails(addUserDto: AddUserDto): Promise<User> {
        const user: User = new User();
        Object.assign({}, user, addUserDto)
        user.aadharNo = addUserDto.aadharNo;
        user.address = addUserDto.address;
        user.age = addUserDto.age;
        user.emailId = addUserDto.emailId;
        user.gender = addUserDto.gender;
        user.infected = addUserDto.infected;
        user.password = addUserDto.password;
        user.phoneNum = addUserDto.phoneNum;
        user.userName = addUserDto.userName;
        user.isActive = true;
        user.updatedAt = "";
        user.updatedTime = "";
        const date = new Date();
        user.createdAt = date.toLocaleDateString();
        user.createTime = date.toLocaleTimeString();
        const login: Login = new Login();
        login.emailId = addUserDto.emailId;
        login.password = user.password;
        login.isActive = true;
        login.updatedAt = "";
        login.updatedTime = "";
        login.createdAt = date.toLocaleDateString();
        login.createTime = date.toLocaleTimeString();
        user.login = login;

        await this.loginService.addLoginDetails(login);
        return await this.userRepository.save(user);
    }
    // async addUserDetails(addUserDto: AddUserDto): Promise<User> {
    // const newEntry  = Object.assign({},AddUserDto,{})

    // }
    /**
     * 
     * @returns all the users from the user table
     */
    async getUserDetails(): Promise<User[]> {
        return await this.userRepository.find();
    }

    async getById(id: number): Promise<User> {
        return await this.userRepository.findOne(id);
    }

    async getByEmailId(emailId: string): Promise<User> {
        return await this.userRepository.findOne(
            { emailId: emailId }
        )
    }
    /**
     * 
     * @param addUserDto accepts object of addUserDto
     * @param id accepts id of type number
     * @returns the updated details of the user
     */
    async updateUserDetails(addUserDto: AddUserDto, id: number): Promise<any> {
        const data = this.getById(id);
        const user: User = new User();

        user.aadharNo = addUserDto.aadharNo;
        user.address = addUserDto.address;
        user.age = addUserDto.age;
        user.emailId = addUserDto.emailId;
        user.gender = addUserDto.gender;
        user.infected = addUserDto.infected;
        user.password = addUserDto.password;
        user.phoneNum = addUserDto.phoneNum;
        user.userName = addUserDto.userName;
        user.isActive = true;
        user.createTime = (await data).createTime;
        user.createdAt = (await data).createdAt;
        const date = new Date();
        user.updatedAt = date.toLocaleDateString();
        user.updatedTime = date.toLocaleTimeString();
        user.id = id

        const result = await this.loginService.getByEmailId(user.emailId);
        result.password = addUserDto.password;
        result.emailId = result.emailId;
        result.updatedTime = date.toLocaleTimeString();
        result.updatedAt = date.toLocaleTimeString();
        result.user = user;
        this.loginRepository.update(result.id, result);
        return await this.userRepository.update(user.id, user);

    }


    /**
     * 
     * @param id recives number as input
     * @returns a message if user gets deleted succesfully
     */
    async logicalUserDelete(id: number): Promise<string> {
        const userResult = await this.getById(id);
        const date = new Date();
        userResult.isActive = false;
        const loginResult = await this.loginService.getByEmailId(userResult.emailId);
        userResult.updatedAt = date.toLocaleTimeString();
        userResult.updatedTime = date.toLocaleDateString();
        this.userRepository.update(id, userResult);
        loginResult.user = userResult;
        this.loginService.updateLoginDetails(loginResult.id, loginResult);
        console.log(userResult);
        return "User Details Deleted Successfully";
    }


}
