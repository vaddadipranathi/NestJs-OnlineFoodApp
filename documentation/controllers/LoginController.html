import { Body, Controller, Get, Post, Req, Res } from '@nestjs/common';
import { LoginService } from './login.service';
import { Response, Request } from 'express';
import { ApiTags } from '@nestjs/swagger';

@Controller('login')
// @ApiBearerAuth()
@ApiTags('login-token')
export class LoginController {
  constructor(private readonly loginService: LoginService) {}

  /**
   *
   * @param emailId recives string type of input
   * @param password recives string type of input
   * @param response recives response object of input
   * @returns jwt token
   */
  @Post('/loginJwt')
  login(
    @Body('emailId') emailId: string,
    @Body('password') password: string,
    @Res({ passthrough: true }) response: Response,
  ): Promise<string> {
    return this.loginService.loginJwtToken(emailId, password, response);
  }

  /**
   *
   * @param request recives request object of input
   * @returns user object of current logged in user
   */
  @Get('/getUser')
  async user(@Req() request: Request) {
    return await this.loginService.getLogggedInUser(request);
  }

  /**
   *
   * @param response recives response object of input
   * @returns a message if user sucesfully log out
   */
  @Post('logout')
  async logout(@Res({ passthrough: true }) response: Response) {
    response.clearCookie('jwt');
    return {
      message: 'successfully logged out',
    };
  }
}
