import { Body, Controller, Delete, Get, Param, Post, Put } from '@nestjs/common';
import { ApiTags } from '@nestjs/swagger';
import { AddVaccineDetailsDto } from 'src/entity/dto/addVaccineDetails.dto';
import { VaccineDetails } from 'src/entity/typeorm/vaccinationData.entity';
import { VaccineDetailsService } from './vaccination-data.service';

/**
 * starting point of VaccineDetailsController
 */
@Controller('VaccineDetails')
@ApiTags('VaccineDetails')
export class VaccineDetailsController {
  constructor(private readonly vaccineDetailsService: VaccineDetailsService,
  ) { }

  /**
 * 
 * @param vaccineDetails accepts objects of vaccineDetails
 * @param emailId accepts emailId of type string
 * @returns the string, if vaccinedetails are added successfully
 */
  @Post('/:emailId')
  async addVaccineDetailsDto(
    @Body() vaccineDetails: AddVaccineDetailsDto,
    @Param('emailId') emailId: string,
  ): Promise<string> {
    return await this.vaccineDetailsService.addVaccineDetails(
      vaccineDetails,
      emailId,
    );
  }

  /**
     * 
     * @returns all the vaccinationDetails from vaccination table
     */
  @Get()
  async getVaccineData(): Promise<VaccineDetails[]> {
    return await this.vaccineDetailsService.getVaccineData();
  }


  /**
    * 
    * @param date accepts date of type date
    * @returns the vaccineData based on the date
    */

  @Get('/getOnDate/:date')
  async getVaccineDataOnDate(
    @Param('date') date: Date,
  ): Promise<VaccineDetails[]> {
    return this.vaccineDetailsService.getVaccineDataOnDate(date);
  }

  @Delete('/:id')
  logicalUserDelete(@Param('id') id: number): Promise<string> {
    return this.vaccineDetailsService.logicalUserDelete(id);
  }
}
