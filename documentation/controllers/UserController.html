import { Body, Controller, Delete, Get, Param, Post, Put } from '@nestjs/common';
import { ApiTags } from '@nestjs/swagger';
import { AddUserDto } from 'src/entity/dto/addUser.dto';
import { User } from 'src/entity/typeorm/user.emtity';
import { UserService } from './user.service';

/**
 * starting of the user controller
 */
@Controller('user')
@ApiTags('user')
export class UserController {
  constructor(private readonly userService: UserService) { }

  /**
    * 
    * @param addUserDto accepts objects of adduserDto
    * @returns user object , if user is added Successfully
    */
  @Post()
  addUserDetails(@Body() addUserDto: AddUserDto): Promise<User> {
    return this.userService.addUserDetails(addUserDto);
  }

  /**
 * 
 * @returns all the users from the user table
 */

  @Get('/getAllUsers')
  getUserDetails(): Promise<User[]> {
    return this.userService.getUserDetails();
  }

  /**
 * 
 * @param addUserDto accepts object of addUserDto
 * @param id accepts id of type number
 * @returns the updated details of the user
 */
  @Put('/updateUserDetails/:id')
  updateUserDetails(
    @Body() addUserDto: AddUserDto,
    @Param('id') id: number,
  ): Promise<any> {
    return this.userService.updateUserDetails(addUserDto, id);
  }

  /**
   * 
   * @param id recives number as input
   * @returns a message if user gets deleted succesfully
   */
  @Delete('deleteUser/:id')
  logicalUserDelete(@Param('id') id: number): Promise<string> {
    return this.userService.logicalUserDelete(id);
  }
}
